    let currentQ = 0;
    let score = 0;
    let timeLeft = 3600; 
    let timer;
    let studentName = "";

    // Захист від гарячих клавіш
    window.addEventListener('keydown', e => {
        if (e.ctrlKey && (e.key === 'c' || e.key === 'v' || e.key === 'u' || e.key === 's')) e.preventDefault();
    });

    function startTest() {
        const nameInput = document.getElementById('student-name').value.trim();
        if (nameInput.length < 5) { alert("Будь ласка, введіть повне Прізвище та Ім'я!"); return; }
        studentName = nameInput;
        document.getElementById('screen-start').classList.remove('active');
        document.getElementById('screen-test').classList.add('active');
        startTimer();
        renderQuestion();
    }

    function startTimer() {
        timer = setInterval(() => {
            timeLeft--;
            const m = Math.floor(timeLeft / 60);
            const s = timeLeft % 60;
            document.getElementById('timer-display').innerText = `${m}:${s < 10 ? '0' : ''}${s}`;
            if (timeLeft <= 0) finishTest();
        }, 1000);
    }

    function renderQuestion() {
        const data = questions[currentQ];
        const container = document.getElementById('question-container');
        document.getElementById('q-number').innerText = `Питання ${currentQ + 1} з ${questions.length}`;
        document.getElementById('progress').style.width = `${(currentQ / questions.length) * 100}%`;
        
        let html = `<div class="q-text">${data.q}</div>`;

        if (data.type === 'single') {
            html += `<div class="options-grid">` + data.options.map((o, i) => `<div class="option" onclick="selectSingle(${i})">${o}</div>`).join('') + `</div>`;
        } else if (data.type === 'multiple') {
            html += `<p style="color:var(--accent-color); font-size:0.8rem; margin-bottom:10px;">* Оберіть декілька варіантів</p><div class="options-grid">` + data.options.map((o, i) => `<div class="option" onclick="selectMulti(${i})">${o}</div>`).join('') + `</div>`;
        } else if (data.type === 'match') {
            html += `<p style="color:var(--accent-color); font-size:0.8rem; margin-bottom:10px;">* Встановіть логічні пари</p>` + data.left.map((l, i) => `
                <div class="match-row">
                    <div class="match-item">${l}</div>
                    <div style="text-align:center">&rarr;</div>
                    <select id="m-${i}"><option value="">---</option>${data.right.map((r, ri) => `<option value="${ri}">${r}</option>`).join('')}</select>
                </div>
            `).join('');
        }
        container.innerHTML = html;
        window.tempAnswer = null;
        window.tempAnswers = [];
    }

    function selectSingle(i) {
        document.querySelectorAll('.option').forEach(el => el.classList.remove('selected'));
        document.querySelectorAll('.option')[i].classList.add('selected');
        window.tempAnswer = i;
    }

    function selectMulti(i) {
        const el = document.querySelectorAll('.option')[i];
        el.classList.toggle('selected');
        const idx = window.tempAnswers.indexOf(i);
        if (idx > -1) window.tempAnswers.splice(idx, 1);
        else window.tempAnswers.push(i);
    }

    function nextQuestion() {
        const data = questions[currentQ];
        let p = 0;

        if (data.type === 'single') {
            if (window.tempAnswer === null) { alert("Оберіть відповідь!"); return; }
            if (window.tempAnswer === data.correct) p = data.points;
        } else if (data.type === 'multiple') {
            if (window.tempAnswers.length === 0) { alert("Оберіть відповіді!"); return; }
            const isMatch = window.tempAnswers.length === data.correct.length && window.tempAnswers.every(v => data.correct.includes(v));
            if (isMatch) p = data.points;
        } else if (data.type === 'match') {
            let matches = 0;
            for(let i=0; i<data.left.length; i++) {
                const val = document.getElementById(`m-${i}`).value;
                if (val === "") { alert("Заповніть всі відповідності!"); return; }
                if (parseInt(val) === data.correct[i]) matches++;
            }
            p = (matches / data.left.length) * data.points;
        }

        score += p;
        currentQ++;
        if (currentQ < questions.length) renderQuestion();
        else finishTest();
    }

    function finishTest() {
        clearInterval(timer);
        document.getElementById('screen-test').classList.remove('active');
        document.getElementById('screen-result').classList.add('active');
        
        const finalScore = Math.round(score * 10) / 10;
        const timeTaken = 3600 - timeLeft;
        const m = Math.floor(timeTaken / 60);
        const s = timeTaken % 60;

        document.getElementById('cert-name').innerText = studentName;
        document.getElementById('cert-score').innerText = finalScore;
        document.getElementById('cert-time').innerText = `Дата: ${new Date().toLocaleDateString()} | Витрачено часу: ${m} хв ${s} сек`;
        document.getElementById('cert-id').innerText = "M3-" + Math.random().toString(36).substr(2, 6).toUpperCase();
        
        document.getElementById('result-info').innerHTML = `Студент: <b>${studentName}</b>. <br> Результат: <b>${finalScore}</b> з 40 балів.`;
    }
